import random import time class JuegoNarcoDispatch: def __init__(self): self.dinero = 500 self.reputacion = 50 self.sospecha_policial = 10 self.mercancia = 20 def mostrar_estado(self): print(f"\n--- ESTADO ACTUAL ---") print(f"Efectivo: ${self.dinero} | Reputación: {self.reputacion} | Sospecha: {self.sospecha_policial}%") print(f"Stock: {self.mercancia} unidades") print("----------------------") def mision_reparto(self): print("\n[ALERTA] Los 'Salamanca' necesitan un envío en el muelle.") print("1. Enviar mensajero rápido (Riesgo alto de policía)") print("2. Enviar camión de pollos (Lento pero seguro)") opcion = input("Elige tu jugada (1/2): ") if opcion == "1": exito = random.randint(1, 100) > 40 if exito: ganancia = 1000 self.dinero += ganancia self.sospecha_policial += 15 print(f"¡ÉXITO! Cobraste ${ganancia}, pero la poli te tiene en la mira.") else: self.sospecha_policial += 30 self.mercancia -= 5 print("¡EMBOSCADA! Perdiste la mercancía y casi te atrapan.") elif opcion == "2": self.dinero += 400 self.sospecha_policial += 2 print("Envío entregado. Poco dinero, pero pasaste bajo el radar.") # Iniciar el juego juego = JuegoNarcoDispatch() while True: juego.mostrar_estado() juego.mision_reparto() time.sleep(2)
